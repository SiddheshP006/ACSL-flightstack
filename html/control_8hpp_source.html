187  /*
188  Implementation of the smooth version of the dead-zone modification of MRAC through a
189  modulation function
190  For reference: E. Lavretsky, K. Wise, "Robust and Adaptive Control", Springer 2013, Sec. 11.2.1
191  PARAMETERS:
192  - delta: constant that must be 0 < delta < 1. delta characterizes the slope of the modulation function.
193  - e_0: constant that must be e_0 > 0. The dead-zone modification stops the adaptation process when
194  the norm of the tracking error becomes smaller than the prescribed value e_0.
195  */
196  template <typename T>
197  inline double deadZoneModulationFunction(const T& e_vector,
198  const double& delta,
199  const double& e_0)
200  {
201  return std::max(0.0, std::min(1.0, (e_vector.norm() - delta * e_0)/((1.0 - delta) * e_0)));
202  }
